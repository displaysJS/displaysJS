            td.start = function() {
              var t=0;
              var counter=null;
              function tick() {
                this.handleTick();
                t+=1;
                if (t > 100) {
                  clearInterval(counter);
                }
              }
              counter = setInterval(tick.bind(this), 1000);
            }

            td.setAsReady();
